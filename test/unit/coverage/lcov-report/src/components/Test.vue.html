<template>
  <div id="my_component">
    <div v-bind:class="{ hidden: !isLoading }">
      <svg xmlns="http://www.w3.org/2000/svg" xmlns:xlink="http://www.w3.org/1999/xlink" style="margin:auto;background:#fff;display:block;" width="200px" height="200px" viewBox="0 0 100 100" preserveAspectRatio="xMidYMid">
        <circle cx="50" cy="50" fill="none" stroke="#1ba4de" stroke-width="4" r="15" stroke-dasharray="70.68583470577033 25.561944901923447">
          <animateTransform attributeName="transform" type="rotate" repeatCount="indefinite" dur="0.9803921568627451s" values="0 50 50;360 50 50" keyTimes="0;1"></animateTransform>
        </circle>
      </svg>
    </div>
 
    <div v-bind:class="{ hidden: isLoading }">
      <h1>{{ result.project_name }}</h1>
    </div>
  </div>
</template>
 
<script>
const Airtable = require('airtable')
 
export default {
  name: 'MyComponent',
  data () {
    return {
      isLoading: true,
      base: null,
      result: {}
    }
  },
  async created () {
    this.result = await this.myLongRunningAsyncFunction()
    this.isLoading = false
  },
  methods: {
    myLongRunningAsyncFunction: async function () {
      let app = this
      let result = {}
      let projectName = 'hackney-repairs__hackney_repairs'
 
      app.base = new Airtable({apiKey: 'keyTA06oep8l3efgd'}).base('appNpQYEz4XOan53g')
 
      let results = await this.findProjectBySlug(projectName)
 
      if (results.length > 0) {
        var project = results[0]
        result.project_name = project.fields['Project Name']
      }
 
      return result
    },
    findProjectBySlug: async function (projectName) {
      return this.base('Projects').select({
        filterByFormula: '({Zendesk Slug}="' + projectName + '")'
      }).all()
    }
  }
}
</script>
